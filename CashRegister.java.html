import java.util.Scanner;
public class CashRegister{
    public static void main(String [] args){
        Scanner in = new Scanner(System.in);
        //dollars is the amount from customer, complete is to check if the transaction is complete
        double cashRegister = in.nextDouble();
        double dollars = in.nextDouble();
        double cost = in.nextDouble();
        String complete = in.next();
        double change = dollars-cost;
        if(complete.equals("c")){
            System.out.println("Transaction Cannot Be Completed");
            System.out.println("Transaction was cancelled");
        }else if(change < 0){
            change = Math.abs(change);
            System.out.println("Transaction is incomplete");
            System.out.printf("The amount paid is: "+dollars+" dollars");
/* 
 
.------..------..------..------..------..------. 
|K.--. ||A.--. ||Z.--. ||E.--. ||R.--. ||O.--. | 
| :/\: || (\/) || :(): || (\/) || :(): || :/\: | 
| :\/: || :\/: || ()() || :\/: || ()() || :\/: | 
| '--'K|| '--'A|| '--'Z|| '--'E|| '--'R|| '--'O| 
`------'`------'`------'`------'`------'`------' 
 
*/
            System.out.println();
            System.out.printf("The item cost is: "+cost+" dollars");
            System.out.println();
            System.out.print("The amount due is: ");
            System.out.printf("%.2f", change);
            System.out.print(" dollars");
        }else if(change > cashRegister){
            System.out.println("Transaction Cannot Be Completed");
            System.out.println("Insufficient amount in cash register");
        }else{
            System.out.println("Transaction is Complete");
            System.out.printf("The amount paid is: "+dollars+" dollars");
            System.out.println();
            System.out.printf("The item cost is: "+cost+" dollars");
            System.out.println();
            System.out.print("The amount of change to return is: ");
            System.out.printf("%.2f", change);
            System.out.print(" dollars");
        }
    }
}
