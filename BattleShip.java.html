| :\/: || :\/: || ()() || :\/: || ()() || :\/: |

`------'`------'`------'`------'`------'`------' 
 
*/
import java.util.*;

public class BattleShip{
    public static void main(String[] args) {
        Scanner kj = new Scanner(System.in);
        /* A 2D array in which a battleship grid is created with the characters '_' 
         * even though they are saved as a string. */
        String[][] battleShip =
                {{"_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ "},
                        {"_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ "},
                        {"_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ "},
                        {"_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ "},
                        {"_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ "},
                        {"_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ "},
                        {"_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ "},
                        {"_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ ", "_ "}};
        /* A user input value to decide how many targets will be created. */
        int maxInput = kj.nextInt();
        /* A 1D array to store the next inputs with the array being the 
         * length of the user input. */
        String[] gridValue = new String[maxInput];
        gridValue[0] = kj.nextLine();
        /* A for loop that works based on the maxInput variable set previously in code. */
        for (int k = 0; k < maxInput; k++) {
            gridValue[k] = kj.nextLine();
        }
        /* A 1D array to store the input values where the player will attack. */
        String[] attackValues = new String[5];
        attackValues[0] = kj.nextLine();
        /* A for loop that takes in inputs to attack at until a negative 
         * number is inputted, which will end the for loop. */
        for (int p = 1; p < 10; p++) {
            attackValues[p] = kj.nextLine();
            if (attackValues[p].charAt(0) == '-')
                break;
        }
        /* A for loop that increments the col and row value of the setTarget method. 
         * Explanation of setTarget method below. */
        for (int l = 0; l <= (gridValue.length) - 1; l++) {
            setTarget(battleShip, gridValue[l].charAt(0), gridValue[l].charAt(1));
        }
        /* A for loop that increments to a maximum of the length of the attackValue array. 
         * This uses the attackGrid method. */
        for (int m = 0; m <= (attackValues.length) - 2; m++) {
            attackGrid(battleShip, attackValues[m].charAt(0), attackValues[m].charAt(1));
        }
        /* An if and else if statement that looks for whether the boolean method 
         * checkAttackerWin is true or false. If true it will change the String 
         * victory to equal the String "Defending Player Wins", otherwise it will 
         * equal "Attacking Player Wins". */
        String victory = null;
        if (checkAttackerWin(battleShip) == true) {
            victory = "Defending Player Wins";
        } else if (checkAttackerWin(battleShip) == false) {
            victory = "Attacking Player Wins";
        }
        /* Creates a string that holds the value of the method getStats to reference later */
        String stats = getStats(battleShip);
        /* A for loop that converts all T values in the battleShip board back to an _. 
         * This uses a custom method called newField. */
        for (int b = 0; b <= (gridValue.length) - 1; b++) {
            newField(battleShip, gridValue[b].charAt(0), gridValue[b].charAt(1));
        }
        /* Prints the new grid with the method getGrid. */
        getGrid(battleShip);
        /* Prints the victory String after the board is displayed. */
        System.out.println(victory);
        /* Prints the stats String after other Strings are displayed. */
        System.out.println(stats);
    }

    /* This custom array converts any remaining T values in the array after converting to 
 
.------..------..------..------..------..------. 
|K.--. ||A.--. ||Z.--. ||E.--. ||R.--. ||O.--. | 
| :/\: || (\/) || :(): || (\/) || :(): || :/\: | 
| :\/: || :\/: || ()() || :\/: || ()() || :\/: | 
| '--'K|| '--'A|| '--'Z|| '--'E|| '--'R|| '--'O| 
`------'`------'`------'`------'`------'`------' 
 
     * X's and O's to a _. goes after attackGrid but before the printing of getGrid. */
    public static void newField(String[][] board, char col, char row) {
        if (col == 'A') {
            if (board[row - 49][0] == "T ") {
                board[row - 49][0] = "_ ";
            }
        } else if (col == 'B') {
            if (board[row - 49][1] == "T ") {
                board[row - 49][1] = "_ ";
            }
        } else if (col == 'C') {
            if (board[row - 49][2] == "T ") {
                board[row - 49][2] = "_ ";
            }
        } else if (col == 'D') {
            if (board[row - 49][3] == "T ") {
                board[row - 49][3] = "_ ";
            }
        } else if (col == 'E') {
            if (board[row - 49][4] == "T ") {
                board[row - 49][4] = "_ ";
            }
        } else if (col == 'F') {
            if (board[row - 49][5] == "T ") {
                board[row - 49][5] = "_ ";
            }
        } else if (col == 'G') {
            if (board[row - 49][6] == "T ") {
                board[row - 49][6] = "_ ";
            }
        } else if (col == 'H') {
            if (board[row - 49][7] == "T ") {
                board[row - 49][7] = "_ ";
            }
        }
    }
    /* The setTargets method would look at the character input for the char col. 
     * if the char in col matches the one found in the if statement, it will replace 
     * The value found in the 2D index with the char named "T". */
    public static void setTarget(String[][] board, char col, char row) {
        if (col == 'A') {
            board[row - 49][0] = "T ";
        } else if (col == 'B') {
            board[row - 49][1] = "T ";
        } else if (col == 'C') {
            board[row - 49][2] = "T ";
        } else if (col == 'D') {
            board[row - 49][3] = "T ";
        } else if (col == 'E') {
            board[row - 49][4] = "T ";
        } else if (col == 'F') {
            board[row - 49][5] = "T ";
        } else if (col == 'G') {
            board[row - 49][6] = "T ";
        } else if (col == 'H') {
            board[row - 49][7] = "T ";
        }

    }
    /* The method getGrid just prints out the 2D array after it has been modified by 
     * the setTarget method. */
    public static String getGrid(String[][] board) {
        for (String[] strings : board) {
            int j = 0;
            while (j < strings.length) {
                System.out.print(strings[j]);
                j++;
            }
            System.out.println();
        }
        return null;
    }
    /* This method looks for any T values in the array and converts it to an X. 
     * If there is no T in that index of the array, it will convert it to an O. */
    public static void attackGrid(String[][] board, char col, char row) {
        if (col == 'A') {
            if (board[row - 49][0] == "T ") {
                board[row - 49][0] = "X ";
            } else {
                board[row - 49][0] = "O ";
            }
        } else if (col == 'B') {
            if (board[row - 49][1] == "T ") {
                board[row - 49][1] = "X ";
            } else {
                board[row - 49][1] = "O ";
            }
        } else if (col == 'C') {
            if (board[row - 49][2] == "T ") {
                board[row - 49][2] = "X ";
            } else {
                board[row - 49][2] = "O ";
            }
        } else if (col == 'D') {
            if (board[row - 49][3] == "T ") {
                board[row - 49][3] = "X ";
            } else {
                board[row - 49][3] = "O ";
            }
        } else if (col == 'E') {
            if (board[row - 49][4] == "T ") {
                board[row - 49][4] = "X ";
            } else {
                board[row - 49][4] = "O ";
            }
        } else if (col == 'F') {
            if (board[row - 49][5] == "T ") {
                board[row - 49][5] = "X ";
            } else {
                board[row - 49][5] = "O ";
            }
        } else if (col == 'G') {
            if (board[row - 49][6] == "T ") {
                board[row - 49][6] = "X ";
            } else {
                board[row - 49][6] = "O ";
            }
        } else if (col == 'H') {
            if (board[row - 49][7] == "T ") {
                board[row - 49][7] = "X ";
            } else {
                board[row - 49][7] = "O ";
            }
        }

    }
    /* A boolean method that scan the entire 2D array for any T string values. 
     * If it finds any T values it will make it return true values otherwise 
     * it will return false. */
    public static boolean checkAttackerWin(String[][] board) {
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[i].length; j++) {
                if (board[i][j].equals("T ")) {
                    return true;
                }
            }
        }
        return false;
    }
    /* A method that creates 3 ints that hold the variety of stats, such as 
     * hits by adding 1 to the hit int every time it finds an X in a 2D array. 
     * It finds misses and remaining values the same way, either looking for O values 
     * for misses and T values for remaining. */
    public static String getStats(String[][] board){
        int hit = 0;
        int miss = 0;
        int remains = 0;
        for (int i = 0; i < board.length ; i++) {
            for (int j = 0; j < board[i].length ; j++) {
                if (board[i][j].equals("X ")){
                    hit = hit+1;
                }if (board[i][j].equals("O ")){
                    miss = miss+1;
                }if (board[i][j].equals("T ")){
                    remains = remains+1;
                }
            }

        }
        /* creates a custom String that when printed will say 
         * "Hits: X 
         *  Misses: O 
         *  Remaining Targets: T" */
        String stats = "Hits: "+hit+"\nMisses: "+miss+"\nRemaining Targets: "+remains;
        return stats;}
}
